Smart Display Templates

Intelligent patterns for displaying contextual recommendations that help your users while generating revenue.

What This Guide Is Aboutâ€‹

This documentation provides reference patterns for intelligently displaying sponsored recommendations in your application. Think of these as proven strategies you can adapt, not rigid rules you must follow.

Your Control, Your Choiceâ€‹

You decide everything:

  • When to fetch recommendations (or not at all)
  • How to display them (subtle text, rich cards, progressive disclosure)
  • Whether to show them (based on your own logic and user preferences)
  • The experience you want to create for your users

These templates are like a cookbook - you can follow recipes exactly, modify them to taste, or just use them for inspiration.

Core Philosophy: Help First, Monetize Secondâ€‹

The best recommendations:

  1. Solve real user problems - Show suggestions when users express needs
  2. Feel natural - Integrate smoothly into conversation flow
  3. Respect user attention - Never interrupt or annoy
  4. Add value - Even sponsored content should be genuinely helpful

Placement Decision Engineâ€‹

Advanced Placement Logicâ€‹

Sophisticated decision-making for optimal ad placement:

interface ConversationContext {
  messageCount: number;
  lastAdShownAt?: number;
  isOnboarding: boolean;
  hasRecentError: boolean;
  hasCommercialIntent: boolean;
  isTaskComplete: boolean;
  conversationPaused: number;
  userSentiment: 'positive' | 'neutral' | 'negative';
  topicCategory: string;
  engagementLevel: number; // 0-100
  timeOfDay: 'morning' | 'afternoon' | 'evening' | 'night';
  deviceType: 'mobile' | 'desktop' | 'tablet';
}

class SmartPlacementEngine {
  private readonly INTENT_KEYWORDS = [
    'need', 'looking for', 'want', 'interested in', 'help with',
    'recommend', 'suggest', 'best', 'compare', 'alternative'
  ];

  shouldShowAd(context: ConversationContext): PlacementDecision {
    // Hard blocks - never show ads
    if (this.isBlockedScenario(context)) {
      return { show: false, reason: 'blocked_scenario' };
    }

    // Calculate placement score
    const placementScore = this.calculatePlacementScore(context);

    if (placementScore < 30) {
      return { show: false, reason: 'low_score' };
    }

    // Determine optimal placement type
    const placement = this.determineOptimalPlacement(context, placementScore);

    return {
      show: true,
      placement: placement.type,
      priority: placement.priority,
      score: placementScore,
      timing: placement.timing
    };
  }

  private isBlockedScenario(context: ConversationContext): boolean {
    return (
      context.isOnboarding ||
      context.hasRecentError ||
      context.messageCount < 3 ||
      context.userSentiment === 'negative' ||
      (context.lastAdShownAt &&
       context.messageCount - context.lastAdShownAt < this.getMinMessageGap(context))
    );
  }

  private getMinMessageGap(context: ConversationContext): number {
    // Dynamic gap based on engagement
    if (context.engagementLevel > 80) return 4;
    if (context.engagementLevel > 50) return 5;
    return 7;
  }

  private calculatePlacementScore(context: ConversationContext): number {
    let score = 0;

    // Commercial intent (0-40 points)
    if (context.hasCommercialIntent) score += 40;

    // Engagement level (0-20 points)
    score += context.engagementLevel * 0.2;

    // Timing factors (0-20 points)
    if (context.conversationPaused > 3000) score += 10;
    if (context.isTaskComplete) score += 10;

    // Context quality (0-20 points)
    if (context.userSentiment === 'positive') score += 10;
    if (this.isPrimeTime(context.timeOfDay)) score += 10;

    return Math.min(score, 100);
  }

  private determineOptimalPlacement(
    context: ConversationContext,
    score: number
  ): PlacementConfig {
    // High intent - immediate inline
    if (score > 70 && context.hasCommercialIntent) {
      return {
        type: 'inline_contextual',
        priority: 'high',
        timing: 'immediate'
      };
    }

    // Task complete - natural follow-up
    if (context.isTaskComplete) {
      return {
        type: 'post_response',
        priority: 'medium',
        timing: 'delayed_1500ms'
      };
    }

    // Conversation pause - gentle suggestion
    if (context.conversationPaused > 3000) {
      return {
        type: 'natural_break',
        priority: 'low',
        timing: 'on_pause'
      };
    }

    // Default - subtle sidebar
    return {
      type: 'sidebar_suggestion',
      priority: 'low',
      timing: 'delayed_2000ms'
    };
  }

  private isPrimeTime(timeOfDay: string): boolean {
    return timeOfDay === 'evening' || timeOfDay === 'afternoon';
  }

  // Intent detection helper
  detectCommercialIntent(message: string): boolean {
    const lowercased = message.toLowerCase();
    return this.INTENT_KEYWORDS.some(keyword => lowercased.includes(keyword));
  }
}

3. Natural Break Placementâ€‹

When to use: During conversation pauses or topic transitions

class NaturalBreakDetector {
  detectBreak(messages: Message[]): boolean {
    const lastMessage = messages[messages.length - 1];
    const timeSinceLastMessage = Date.now() - lastMessage.timestamp;

    // Natural pause detected
    if (timeSinceLastMessage > 3000) return true;

    // Topic transition detected
    if (this.isTopicChange(messages)) return true;

    // Task completion detected
    if (lastMessage.text.match(/done|completed|finished|thanks/i)) return true;

    return false;
  }
}

5. Progressive Disclosure Patternâ€‹

When to use: When you want your AI to naturally reveal recommendations over multiple messages, giving users control over engagement.

This pattern lets AI agents introduce sponsored content conversationally, only revealing details when users show interest. Payment triggers when users engage with the initial teaser.

Multi-Message Revelation Flowâ€‹

class ProgressiveDisclosureManager {
  private pendingRecommendation: Ad | null = null;
  private disclosureState: 'idle' | 'teased' | 'revealed' = 'idle';

  async handleUserMessage(message: string, context: Context) {
    // Step 1: Fetch recommendation if relevant
    if (this.detectIntent(message) && !this.pendingRecommendation) {
      this.pendingRecommendation = await client.decideFromContext({
        userMessage: message,
        conversationHistory: context.history
      });
    }

    // Step 2: Natural conversation response
    const aiResponse = await generateResponse(message);
    await displayMessage(aiResponse);

    // Step 3: Decide if/how to introduce recommendation
    if (this.pendingRecommendation && this.shouldIntroduce(context)) {
      await this.startProgressiveDisclosure();
    }
  }

  async startProgressiveDisclosure() {
    // Teaser message - natural and conversational
    await displayMessage({
      content: "By the way, I found a 20% discount that might interest you. Would you like to hear about it?",
      type: 'ai_suggestion'
    });

    this.disclosureState = 'teased';

    // Track impression - payment triggers here if user engages
    await client.trackImpression(this.pendingRecommendation.tracking_token);
  }

  async handleUserResponse(response: string) {
    if (this.disclosureState !== 'teased') return;

    const interested = this.detectInterest(response);

    if (interested) {
      // User said yes - reveal full details
      await this.revealRecommendation();
    } else if (this.detectDisinterest(response)) {
      // User said no - acknowledge and move on
      await displayMessage("No problem! Let me know if you change your mind.");
      this.resetDisclosure();
    } else {
      // Ambiguous response - continue conversation naturally
      // Keep recommendation pending for potential later reveal
    }
  }

  async revealRecommendation() {
    const rec = this.pendingRecommendation;

    await displayMessage({
      content: `Great! ${rec.creative.title}\n\n${rec.creative.body}\n\nUse code: ${rec.creative.promo_code}\n\nHere's the link: ${rec.click_url}`,
      metadata: {
        sponsored: true,
        sponsor: rec.disclosure.sponsor_name
      }
    });

    this.disclosureState = 'revealed';
    await client.trackClick(rec.tracking_token);
  }

  detectInterest(response: string): boolean {
    const positive = ['yes', 'sure', 'tell me', 'interested', 'yeah', 'ok', 'please'];
    return positive.some(word => response.toLowerCase().includes(word));
  }

  detectDisinterest(response: string): boolean {
    const negative = ['no', 'not interested', 'skip', 'pass', 'later'];
    return negative.some(word => response.toLowerCase().includes(word));
  }
}

Payment Triggers in Progressive Disclosureâ€‹

class PaymentTriggerManager {
  // Payment happens at engagement, not reveal
  async trackProgressiveDisclosure(recommendation: Ad, userAction: string) {
    switch(userAction) {
      case 'teaser_shown':
        // No payment yet - just tracking
        await this.analytics.track('teaser_displayed', recommendation.id);
        break;

      case 'user_interested':
        // PAYMENT TRIGGER - User engaged with teaser
        await client.trackClick(recommendation.tracking_token);
        await this.analytics.track('progressive_disclosure_success', {
          recommendation_id: recommendation.id,
          payout: recommendation.payout
        });
        break;

      case 'user_not_interested':
        // No payment - user declined
        await this.analytics.track('progressive_disclosure_declined', recommendation.id);
        break;

      case 'link_clicked':
        // Additional tracking (payment already triggered)
        await this.analytics.track('final_click_through', recommendation.id);
        break;
    }
  }
}

Best Practices for Progressive Disclosureâ€‹

Do's:

  • âœ… Complete the user's request BEFORE introducing recommendations
  • âœ… Make teasers feel like helpful suggestions, not sales pitches
  • âœ… Accept "no" gracefully without arguing
  • âœ… Keep recommendations relevant to the ongoing conversation
  • âœ… Use natural language that matches your AI's personality

Don'ts:

  • âŒ Lead with the recommendation before answering
  • âŒ Push if user shows disinterest
  • âŒ Reveal full details without permission
  • âŒ Make the teaser feel like an interruption
  • âŒ Hide the sponsored nature when revealing

class ProgressiveDisclosure {
  calculateAdFrequency(session: Session): number {
    const baseInterval = 10; // messages
    const engagementMultiplier = Math.min(session.engagementScore / 100, 1);
    const sessionLengthBonus = Math.min(session.messageCount / 50, 0.5);

    // More engaged users see slightly more relevant ads
    const adjustedInterval = baseInterval * (1 - engagementMultiplier * 0.3);

    // Longer sessions get better spacing
    return Math.ceil(adjustedInterval + sessionLengthBonus * 5);
  }
}

class AutonomousRecommendationAgent {
  private readonly strategies = {
    highIntent: {
      threshold: 0.8,
      approach: 'immediate',
      displayFormat: 'inline_with_response'
    },
    mediumIntent: {
      threshold: 0.5,
      approach: 'progressive',
      displayFormat: 'teaser_then_reveal'
    },
    lowIntent: {
      threshold: 0.3,
      approach: 'deferred',
      displayFormat: 'end_of_conversation'
    }
  };

  async processWithRecommendations(userInput: string) {
    // 1. Generate primary response
    const response = await this.generateResponse(userInput);

    // 2. Analyze for recommendation opportunity
    const analysis = await this.analyzeOpportunity(userInput, response);

    // 3. Fetch recommendation if appropriate
    if (analysis.score < 0.3) {
      return response; // No recommendation
    }

    const recommendation = await this.fetchRecommendation(userInput);
    if (!recommendation) return response;

    // 4. Choose strategy based on score
    const strategy = this.selectStrategy(analysis.score);

    // 5. Format according to strategy
    return this.formatWithStrategy(response, recommendation, strategy);
  }

  selectStrategy(score: number): DisplayStrategy {
    if (score >= 0.8) return this.strategies.highIntent;
    if (score >= 0.5) return this.strategies.mediumIntent;
    return this.strategies.lowIntent;
  }

  formatWithStrategy(
    response: string,
    recommendation: Ad,
    strategy: DisplayStrategy
  ): FormattedResponse {
    switch(strategy.approach) {
      case 'immediate':
        return {
          content: `${response}\n\nðŸ’¡ Relevant offer: ${recommendation.creative.title}\n${recommendation.creative.body}\n[${recommendation.creative.cta}](${recommendation.click_url})\n*(Sponsored)*`
        };

      case 'progressive':
        return {
          content: response,
          followUp: {
            delay: 2000,
            content: `I found a ${recommendation.creative.teaser || 'relevant offer'}. Would you like details?`,
            onPositive: () => this.revealRecommendation(recommendation)
          }
        };

      case 'deferred':
        return {
          content: response,
          deferredRecommendation: recommendation,
          showAfterMessages: 2
        };
    }
  }
}

    return null;
  }

  formatProgressiveResponse(aiResponse, recommendation) {
    return `${aiResponse}

---
ðŸ’¡ By the way, ${recommendation.creative.teaser || 'I found something that might help'}.

Reply with "yes" if you'd like details, or we can continue our conversation.`;
  }

  hasCommercialIntent(message) {
    const intents = [
      /\b(need|want|looking for|recommend|suggest)\b/i,
      /\b(best|top|good|cheap|affordable)\b/i,
      /\b(where|how|what).*(buy|get|find)\b/i
    ];
    return intents.some(pattern => pattern.test(message));
  }
}
